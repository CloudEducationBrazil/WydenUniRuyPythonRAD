import mysql.connector

conn = mysql.connector.connect(
    host="tramway.proxy.rlwy.net",
    port=55014,
    user="root",
    password="jSaQXTanXzgrKhKubXrqLKJuWorAPBUG",
    database="railway"
)

cur = conn.cursor()

id_user = int(input("Digite o ID do usuário que deseja atualizar: "))

# Verifica se existe antes de pedir os outros dados
cur.execute("SELECT id, name, idade FROM `user` WHERE id = %s;", (id_user,))
resultado = cur.fetchone()

if resultado is None:
    print("ID não encontrado. Nenhum usuário com esse ID.")
else:
    print("Usuário encontrado:", resultado)

    novo_nome = input("Digite o novo nome: ")
    nova_idade = int(input("Digite a nova idade: "))

    sql = "UPDATE `user` SET name = %s, idade = %s WHERE id = %s;"
    cur.execute(sql, (novo_nome, nova_idade, id_user))
    conn.commit()

    print("Atualizado com sucesso!")

cur.close()
conn.close()

import mysql.connector

conn = mysql.connector.connect(
    host="tramway.proxy.rlwy.net",
    port=55014,
    user="root",
    password="jSaQXTanXzgrKhKubXrqLKJuWorAPBUG",
    database="railway"
)

cur = conn.cursor()

id_user = int(input("Digite o ID do usuário que deseja deletar: "))

# Verifica se existe antes de deletar
cur.execute("SELECT id, name, idade FROM `user` WHERE id = %s;", (id_user,))
resultado = cur.fetchone()

if resultado is None:
    print("ID não encontrado. Nenhum usuário com esse ID.")
else:
    print("Usuário encontrado:", resultado)

    confirmar = input("Tem certeza que deseja deletar? (s/n): ").lower()

    if confirmar == "s":
        cur.execute("DELETE FROM `user` WHERE id = %s;", (id_user,))
        conn.commit()
        print("Usuário deletado com sucesso!")
    else:
        print("Operação cancelada.")

cur.close()
conn.close()

print("Conexão encerrada")
